/* ── TYPING ANIMATION ── */
const phrases = [
  "You are my sunshine 🌸",
  "My favorite sister 💝",
  "Born to shine ✨",
  "My heart's best part 🌹",
  "Wishing you the world 🌍",
];
let pi = 0, ci = 0, deleting = false;
const typingEl = document.getElementById('typing');
function type() {
  const current = phrases[pi];
  if (!deleting) {
    typingEl.textContent = current.slice(0, ++ci);
    if (ci === current.length) { deleting = true; setTimeout(type, 2200); return; }
  } else {
    typingEl.textContent = current.slice(0, --ci);
    if (ci === 0) { deleting = false; pi = (pi + 1) % phrases.length; }
  }
  setTimeout(type, deleting ? 55 : 90);
}
setTimeout(type, 2500);

/* ── SPARKLE ON CLICK ── */

  for (let i = 0; i < 6; i++) {
    const sp = document.createElement('div');
    sp.className = 'sparkle';
    const emojis = ['✨','⭐','💫','🌟','💖','🌸'];
    sp.textContent = emojis[Math.floor(Math.random()*emojis.length)];
    sp.style.cssText = `
      left:${e.clientX + (Math.random()*60-30)}px;
      top:${e.clientY + (Math.random()*60-30)}px;
      font-size:${14+Math.random()*16}px;
    `;
    document.body.appendChild(sp);
    setTimeout(() => sp.remove(), 700);
  }
});

/* ── INTERSECTION OBSERVER FOR WISH CARDS ── */
const cards = document.querySelectorAll('.wish-card');
cards.forEach((c, i) => {
  c.style.opacity = '0';
  c.style.transform = 'translateY(30px)';
  c.style.transition = `opacity .6s ease ${i*.1}s, transform .6s ease ${i*.1}s`;
});
const obs = new IntersectionObserver(entries => {
  entries.forEach(e => {
    if (e.isIntersecting) {
      e.target.style.opacity = '1';
      e.target.style.transform = 'translateY(0)';
    }
  });
}, { threshold: .15 });
cards.forEach(c => obs.observe(c));
